#version 430

layout(location = 0) in vec4 position;

layout(location = 0) uniform mat4 Projection;
layout(location = 1) uniform mat4 View;
layout(location = 2) uniform mat4 Model;

out vec4 vPosition;

void main() 
{
	vPosition = position;
	
	
	gl_Position = Projection * View * Model * position; 

}
